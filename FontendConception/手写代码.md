# 常见的代码

### 1. 手写bind函数

**bind函数原理**

```js
function test(a,b,c){
	console.log(a,b,c)
	console.log('this',this)
	return 'hello'
}

const result = test(1,10,100)

const boundTest = test.bind({name:'Dennis'},2,20,2000)
const boundResult = boundTest()

console.log('result',result) // 1, 10, 100; this window; hello 
console.log('boundResult',boundResult)// 2, 20, 200; this {name:'Dennis'}; hello
```

**bind手写**

```js
Function.prototype.myBind = function(){ //bind apply call 方法本来就保存在Function原型链上
    const self = this;
    const args = Array.prototype.slice.call(arguments); //转化类数组到数组
    const thisValue = args.shift()
    return function(){
        return self.apply(thisValue,args)
    }
}
```



#### 2. 防抖与节流

**防抖**：防止表单多次提交

```js
function debounce(fn, delay){
    let timer; //用闭包实现变量唯一，通过不断赋值来增加延时
	return function(){
        let that = this;
        let args = arguements;
        clearTimeout(timer)
        timer = setTimeout(function(){
            fn.apply(that, args)
        },delay)
    }
}
```

**节流**：用户滚动改变页面指定事件

```js
function throttle(fn, delay){
    let timer;
    return function(){
        let that = this;
        let args = arguements;
        if(timer) return;
        timer = setTimeout(function(){
            fn.apply(that,args)
            timer= null;
        },delay) 
    }
}
```

#### 4. 深浅拷贝

```js
//simpleClone
function SimpleClone(obj){
    let cloneObj = {}
    
    for(let i in obj){
        cloneObj[i] = obj[i]
    }
    return cloneObj;
}

//deepClone
function deepClone(obj = {}){
    if(typeof obj !== obj || obj === null){
        return obj
    }
    let result;
    if(obj instanceof Array){
        result = []
    }else{
        result = {}
    }
    
    for (let i in obj){
        if(obj.hasOwnProperty(i)){
            result[i] = deepClone(obj[i]);
        }
    }
    return result
}


//deepClone 有些问题
function deepClone(obj,cloneObj){
    cloneObj = cloneObj || {}
    for(let i in obj){
        if(typeof obj[i] === 'object' && obj[i] !== null){
                // cloneObj[i] = obj[i] instanceof Array ? []:{};
            	// cloneObj[i] = Array.isArray(obj[i])?[]:{}
            	cloneObj[i] = Object.prototype.toString.call(obj[i]) === ['object Array']? []:{}
                deepClone(obj[i],cloneObj[i])
            }
            else{
                cloneObj[i] = obj[i];
            }
        }
    }
    return cloneObj;
}
```
